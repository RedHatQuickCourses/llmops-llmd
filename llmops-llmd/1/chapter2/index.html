INFERENCE_URL=$( oc -n openshift-ingress get gateway openshift-ai-inference \ -o jsonpath='{.status.addresses[0].value}' )

LLM=openai/gpt-oss-20b LLM_SVC=${LLM##*/}

PROMPT="Give me a one page explaination of why people would think the earth is flat"

llm_post_data(){ cat <<JSON { "model": "${LLM}", "prompt": "${PROMPT}", "max_tokens": 20000, "temperature": 0.7, "top_p": 0.9 } JSON }

curl -s -X POST http://${INFERENCE_URL}/demo-llm/${LLM_SVC}/v1/completions \ -H "Content-Type: application/json" \ -d "$(llm_post_data)" | jq .choices[0].text
